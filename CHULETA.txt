////////OFERTAS CON AVISOS CON MENSAJE "GENERADO AUTOMATICAMENTE"

    ////para ver las ofertas


        SELECT
            op.*
        FROM
            ofertas_producto AS op
        INNER JOIN
            avisos AS a ON op.id = a.avisoable_id
        WHERE
            a.texto_aviso LIKE '%GENERADO%';

    /////para cambiar las ofertas a monstrar->si que tienen un aviso con la palabra GENERADO

        UPDATE
            ofertas_producto AS op
        INNER JOIN
            avisos AS a ON op.id = a.avisoable_id
        SET
            op.mostrar = 'si'
        WHERE
            a.texto_aviso LIKE '%GENERADO%';

    //eliminar los avisos con la palabra GENERADO

    DELETE FROM avisos WHERE texto_aviso LIKE '%GENERADO%';
